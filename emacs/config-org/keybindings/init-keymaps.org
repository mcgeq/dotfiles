** Line Manipulation
#+BEGIN_SRC emacs-lisp
  ;;; === 行操作 ===
  
  ;; Insert newline above/below
  (global-set-key (kbd "C-c <down>")
                  (lambda ()
                    (interactive)
                    (end-of-line)
                    (newline-and-indent)))
  
  (global-set-key (kbd "C-c <up>")
                  (lambda ()
                    (interactive)
                    (beginning-of-line)
                    (newline)
                    (forward-line -1)
                    (indent-according-to-mode)))
  
  ;; Move lines up/down
  (global-set-key (kbd "M-<up>")
                  (lambda ()
                    (interactive)
                    (transpose-lines 1)
                    (forward-line -2)
                    (indent-according-to-mode)))
  
  (global-set-key (kbd "M-<down>")
                  (lambda ()
                    (interactive)
                    (forward-line 1)
                    (transpose-lines 1)
                    (forward-line -1)
                    (indent-according-to-mode)))
#+END_SRC

  (defun mcg-which-key-show-all-prefixes ()
    "Show all defined prefix key replacements."
    (interactive)
    (with-help-window "*Which-key Prefixes*"
      (princ "Which-key Defined Prefixes:\n\n")
      (dolist (replacement (sort (copy-sequence which-key--prefix-title-alist)

")))
