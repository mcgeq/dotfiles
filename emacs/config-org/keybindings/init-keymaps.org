** Line Manipulation
#+BEGIN_SRC emacs-lisp
  ;;; === 行操作 ===
  
  ;; Insert newline above/below
  (global-set-key (kbd "C-c <down>")
                  (lambda ()
                    (interactive)
                    (end-of-line)
                    (newline-and-indent)))
  
  (global-set-key (kbd "C-c <up>")
                  (lambda ()
                    (interactive)
                    (beginning-of-line)
                    (newline)
                    (forward-line -1)
                    (indent-according-to-mode)))
  
  ;; Move lines up/down
  (global-set-key (kbd "M-<up>")
                  (lambda ()
                    (interactive)
                    (transpose-lines 1)
                    (forward-line -2)
                    (indent-according-to-mode)))
  
  (global-set-key (kbd "M-<down>")
                  (lambda ()
                    (interactive)
                    (forward-line 1)
                    (transpose-lines 1)
                    (forward-line -1)
                    (indent-according-to-mode)))
#+END_SRC

** Scrolling
#+BEGIN_SRC emacs-lisp
  ;;; === 滚动 ===
  
  (defun mcg-scroll-down-third ()
    "Scroll down 1/3 of window height (view later content)."
    (interactive)
    (scroll-up (/ (window-body-height) 3)))
  
  (defun mcg-scroll-up-third ()
    "Scroll up 1/3 of window height (view earlier content)."
    (interactive)
    (scroll-down (/ (window-body-height) 3)))
  
  ;; Fast scrolling with M-n/M-p
  ;; M-n (next) scrolls down to view later content
  ;; M-p (previous) scrolls up to view earlier content
  (global-set-key (kbd "M-n") 'mcg-scroll-down-third)
  (global-set-key (kbd "M-p") 'mcg-scroll-up-third)
#+END_SRC
