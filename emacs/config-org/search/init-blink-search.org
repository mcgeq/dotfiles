** Basic Configuration
#+BEGIN_SRC emacs-lisp
  ;;; === 基本配置 ===
  
  ;; Common directories for quick access
  (setq blink-search-common-directory 
        `(("HOME" "~/")
          ("REPO" ,(concat mcgemacs-root-dir "/site-lisp/extensions"))
          ("CONFIG" ,mcgemacs-config-dir)
          ;; 博客目录：使用新的 Org 配置变量
          ("BLOG" ,(if (featurep 'init-org-vars)
                       mcg-org-blog-directory
                     (concat mcgemacs-blog-dir "/content-org")))
          ("DOWNLOADS" "~/Downloads/")
          ("DOCUMENTS" "~/Documents/")
          ("DESKTOP" "~/Desktop/")))
  
  ;; Python command (ensure correct Python interpreter)
  (setq blink-search-python-command 
        (if (memq system-type '(cygwin windows-nt ms-dos)) 
            "python.exe" 
          "python3"))
#+END_SRC

** Helper Functions
#+BEGIN_SRC emacs-lisp
  ;;; === 辅助函数 ===
  
  (defun mcg/blink-search-grep-current-dir ()
    "Search files in current directory using blink-search."
    (interactive)
    (let ((default-directory (or (when buffer-file-name
                                   (file-name-directory buffer-file-name))
                                 default-directory)))
      (blink-search)))
  
  (defun mcg/blink-search-project-root ()
    "Search files in project root using blink-search."
    (interactive)
    (let* ((project-root (or (and (fboundp 'projectile-project-root)
                                  (projectile-project-root))
                             (and (fboundp 'project-root)
                                  (when-let ((project (project-current)))
                                    (project-root project)))
                             default-directory))
           (default-directory project-root))
      (blink-search)))
  
  (defun mcg/blink-search-buffers ()
    "Search and switch between open buffers.
Tip: Type to filter buffers, or press TAB to see all backends."
    (interactive)
    (blink-search))
  
  (defun mcg/blink-search-recent-files ()
    "Search recent files.
Tip: Type to filter, or press TAB to see all backends."
    (interactive)
    (blink-search))
  
  (defun mcg/blink-search-imenu ()
    "Navigate using IMenu in current buffer."
    (interactive)
    (blink-search))

  ;;   C-c s D - mcg/blink-search-grep-current-dir (current dir files)
  ;;   C-c s P - mcg/blink-search-project-root (project files)
