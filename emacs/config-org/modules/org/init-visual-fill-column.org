** Configuration

#+begin_src emacs-lisp
;;; ============================================================
;;; Configuration
;;; ============================================================

(defcustom mcg-org-enable-visual-fill nil
  "Whether to enable visual fill column."
  :type 'boolean
  :group 'mcg-org)

(defcustom mcg-org-visual-fill-width 88
  "Visual fill column width."
  :type 'integer
  :group 'mcg-org)

(defun mcg--setup-visual-fill-column ()
  "Set up visual-fill-column for org mode."
  (with-eval-after-load 'org
    (when (and mcg-org-enable-visual-fill
               (mcg-require-extension "visual-fill-column" t))
      (setq visual-fill-column-width mcg-org-visual-fill-width)
      (add-hook 'org-mode-hook #'visual-fill-column-mode)
      (mcg-log "Visual fill column configured"))))
#+end_src

** Initialize

#+begin_src emacs-lisp
(mcg--setup-visual-fill-column)
#+end_src
