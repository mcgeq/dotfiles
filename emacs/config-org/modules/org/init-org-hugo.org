  :type 'string
  :group 'mcg-org-hugo)

  "Default file extension for Hugo posts."

** Easy Hugo Configuration

easy-hugo 配置，用于管理 Hugo 博客。

#+begin_src emacs-lisp
;;; ============================================================
;;; Easy Hugo Configuration
;;; ============================================================

(defun mcg-org-hugo-setup-easy-hugo ()
  "Set up easy-hugo for blog management."
  (when (and mcg-org-hugo-enable
             (mcg-require-extension "easy-hugo" t))
    (setq easy-hugo-postdir mcg-org-hugo-postdir
          easy-hugo-default-ext mcg-org-hugo-default-ext
          easy-hugo-url mcg-org-hugo-url
          easy-hugo-basedir (file-name-as-directory mcg-blog-base-directory))
    (mcg-log "Easy Hugo configured: %s" mcg-blog-base-directory)))
#+end_src

** Hugo Capture Templates

Hugo 博客的 org-capture 模板。

#+begin_src emacs-lisp
;;; ============================================================
;;; Hugo Capture Templates
;;; ============================================================

(defun mcg/org-hugo-new-subtree-post-capture-template ()
  "Return org-capture template string for new Hugo post."
  (let* ((date (format-time-string (org-time-stamp-format :long :inactive)))
         (title (read-from-minibuffer "Post Title: "))
         (fname (org-hugo-slug title)))
    (mapconcat #'identity
               `(,(concat "* TODO " title)
                 ":PROPERTIES:"
                 ,(concat ":EXPORT_FILE_NAME: " fname)
                 ,(concat ":EXPORT_DATE: " date)
                 ":END:"
                 "%?\n")
               "\n")))
#+end_src

** Hugo Integration

Hugo 集成配置，包括 capture 模板。

#+begin_src emacs-lisp
;;; ============================================================
;;; Hugo Integration
;;; ============================================================

(defun mcg-org-hugo-setup-capture-templates ()
  "Setup Hugo blog capture templates."
  (when mcg-org-hugo-enable
    ;; Add Hugo post category
    (add-to-list 'org-capture-templates '("h" "Hugo Post"))
    (add-to-list 'org-capture-templates '("hl" "About Development Language"))

    ;; Language-specific templates
    (dolist (topic '(("hla" "ArkTS" "ArkTS")
                     ("hle" "React" "React")
                     ("hlr" "Rust" "Rust")
                     ("hlv" "Vue" "Vue")))
      (add-to-list 'org-capture-templates
                   `(,(nth 0 topic)
                     ,(concat "About " (nth 1 topic))
                     entry
                     (file+headline mcg-org-blog-development-file ,(nth 2 topic))
                     (function mcg/org-hugo-new-subtree-post-capture-template))))

    ;; Emacs template
    (add-to-list 'org-capture-templates
                 '("he" "About Emacs"
                   entry
                   (file+olp mcg-org-blog-emacs-file "Emacs")
                   (function mcg/org-hugo-new-subtree-post-capture-template)))

    (mcg-log "Hugo capture templates configured")))
#+end_src

  "Initialize Hugo blog integration."
  (when mcg-org-hugo-enable

    (mcg-org-hugo-setup-ox-hugo)

(mcg-org-hugo-init)
