(defun mcg--setup-org-appear ()
  "Set up org-appear."
  (when (mcg-require-extension "org-appear" t)
    (org-appear-mode 1)
    (setq org-appear-autosubmarkers t
          org-appear-autolinks t)
    (mcg-log "Org appear configured")))
#+end_src

** Initialize

#+begin_src emacs-lisp
