;;; ============================================================

(defun mcg--setup-org-superstar ()
  "Set up org-superstar-mode."
  (with-eval-after-load 'org
    (when (mcg-require-extension "org-superstar-mode" t)
      ;; Use unified bullets from mcg-org-bullets-alist
      (let ((headline-bullets (car mcg-org-bullets-alist))
            (item-bullets (cadr mcg-org-bullets-alist)))
        (setq org-superstar-leading-bullet ?\s
              org-superstar-special-todo-items t
              org-superstar-item-bullet-alist item-bullets
              org-superstar-headline-bullets-list headline-bullets))

      (custom-set-faces
       '(org-superstar-item ((t (:inherit 'fixed-pitch))))
       '(org-superstar-header-bullet ((t (:height 232 :inherit 'fixed-pitch)))))

      (org-superstar-mode 1)
      (mcg-log "Org superstar configured"))))
#+end_src

#+begin_src emacs-lisp
