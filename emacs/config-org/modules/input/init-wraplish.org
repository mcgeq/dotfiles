** Dependencies
#+BEGIN_SRC emacs-lisp
;;; Dependencies

(require 'core-lib)

#+END_SRC

** Variables
#+BEGIN_SRC emacs-lisp
;;; Variables

(defvar mcg-wraplish-modes
  '(markdown-mode-hook
    markdown-ts-mode-hook
    org-mode-hook
    prog-mode-hook)
  "启用 wraplish 的模式列表。")

#+END_SRC

** Configuration
#+BEGIN_SRC emacs-lisp
;;; Configuration

(defun mcg--setup-wraplish ()
  "设置 wraplish 中英文混排。"
  (when (mcg-require-extension "input/wraplish" 'wraplish)
    (dolist (hook mcg-wraplish-modes)
      (add-hook hook (lambda () (wraplish-mode 1))))
    
    (mcg-log "Configured wraplish")))

#+END_SRC

** Initialize
#+BEGIN_SRC emacs-lisp
;;; Initialize

(mcg--setup-wraplish)

#+END_SRC

** Provide
#+BEGIN_SRC emacs-lisp
(provide 'init-wraplish)
;;; init-wraplish.el ends here
#+END_SRC
