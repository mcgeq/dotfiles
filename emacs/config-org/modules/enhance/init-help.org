** Helpers

#+begin_src emacs-lisp
(defun mcg-help--helpful-available-p ()
  "Return non-nil when helpful is available."
  (require 'helpful nil t))

(defun mcg-help-describe-function ()
  "Describe a function, using helpful when available."
  (interactive)
  (if (and (mcg-help--helpful-available-p)
           (fboundp 'helpful-callable))
      (call-interactively #'helpful-callable)
    (call-interactively #'describe-function)))

(defun mcg-help-describe-variable ()
  "Describe a variable, using helpful when available."
  (interactive)
  (if (and (mcg-help--helpful-available-p)
           (fboundp 'helpful-variable))
      (call-interactively #'helpful-variable)
    (call-interactively #'describe-variable)))

** Initialize

#+begin_src emacs-lisp
(when (and mcg-help-enable-which-key
           (featurep 'which-key))
  (setq which-key-idle-delay mcg-help-which-key-delay)
  (setq which-key-idle-secondary-delay 0.05)
  (setq which-key-show-early-on-C-h t)
  (setq which-key-sort-order 'which-key-key-order-alpha)
  (which-key-mode 1))

(define-prefix-command 'mcg-help-map)
(mcg-bind-key "C-c h" 'mcg-help-map "enhance/help")
(mcg-bind-key "C-c h f" #'mcg-help-describe-function "enhance/help")
(mcg-bind-key "C-c h v" #'mcg-help-describe-variable "enhance/help")
(mcg-bind-key "C-c h k" #'mcg-help-describe-key "enhance/help")
(mcg-bind-key "C-c h o" #'mcg-help-describe-symbol "enhance/help")
(mcg-bind-key "C-c h b" #'describe-bindings "enhance/help")
(mcg-bind-key "C-c h m" #'describe-mode "enhance/help")
(mcg-bind-key "C-c h w" #'where-is "enhance/help")
(mcg-bind-key "C-c h c" #'describe-char "enhance/help")
(mcg-bind-key "C-c h l" #'view-lossage "enhance/help")

(mcg-bind-key "C-h f" #'mcg-help-describe-function "enhance/help")
(mcg-bind-key "C-h v" #'mcg-help-describe-variable "enhance/help")
(mcg-bind-key "C-h k" #'mcg-help-describe-key "enhance/help")
(mcg-bind-key "C-h o" #'mcg-help-describe-symbol "enhance/help")

(mcg-log "help module configured")
