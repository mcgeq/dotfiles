** Setup

#+begin_src emacs-lisp
(defun mcg--setup-pinyinlib ()
  "Setup pinyinlib."
  (if (mcg-require-extension "pinyinlib" t)
      (mcg-log "pinyinlib configured")
    (mcg-log-warning "pinyinlib not available")))
#+end_src

** Initialize

#+begin_src emacs-lisp
(mcg--setup-pinyinlib)
#+end_src

#+begin_src emacs-lisp
