** Setup

#+begin_src emacs-lisp
(defun mcg--setup-embark ()
  "Setup embark."
  (when (mcg-require-extension "embark" t)
    (mcg-bind-key "C-." #'embark-act "completion/embark")
    (mcg-bind-key "C-;" #'embark-dwim "completion/embark")
    (mcg-log "embark configured")))
#+end_src

** Initialize

#+begin_src emacs-lisp
(mcg--setup-embark)
#+end_src
