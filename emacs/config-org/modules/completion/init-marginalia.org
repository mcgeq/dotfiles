** Setup

#+begin_src emacs-lisp
;;; ============================================================
;;; Marginalia Configuration using mcg-simple-module
;;; ============================================================

(mcg-simple-module "marginalia"
    (when (mcg-require-extension "marginalia" t)
      ;; Alignment settings
      (setq marginalia-align 'right
            marginalia-align-offset 0
            marginalia-max-relative-age 3600)
      ;; Enable marginalia-mode
      (marginalia-mode 1)
      ;; Keybinding for cycling annotation levels
      (define-key minibuffer-local-map (kbd "M-A") #'marginalia-cycle))
  (("M-A" . marginalia-cycle)))

;;; ============================================================
;;; Commands
;;; ============================================================

(defun mcg/toggle-marginalia ()
  "Toggle Marginalia mode."
  (interactive)
  (if marginalia-mode
      (progn
        (marginalia-mode -1)
        (message "Marginalia disabled"))
    (marginalia-mode 1)
    (message "Marginalia enabled")))
#+end_src
