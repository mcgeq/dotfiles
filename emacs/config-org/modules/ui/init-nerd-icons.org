** Setup

#+begin_src emacs-lisp
(defun mcg--setup-nerd-icons ()
  "Setup nerd-icons."
  (if (mcg-require-extension "nerd-icons" t)
      (progn
        (when (boundp 'mcg-ui-icon-font-family)
          (setq nerd-icons-font-family mcg-ui-icon-font-family))
        (if (and (boundp 'nerd-icons-font-family)
                 (find-font (font-spec :family nerd-icons-font-family)))
            (mcg-log "nerd-icons configured: %s" nerd-icons-font-family)
          (mcg-log-warning "nerd-icons font missing, run M-x nerd-icons-install-fonts")))
    (mcg-log-warning "nerd-icons not available")))
#+end_src

** Initialize

#+begin_src emacs-lisp

#+end_src
