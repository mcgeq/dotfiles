* Format Module

统一的代码格式化入口。优先使用已经启动的 LSP 能力；没有 LSP 时，按当前 mode 调用本地 formatter。

** Helpers

#+begin_src emacs-lisp
(defvar-local mcg-format--inhibit-on-save nil
  "Non-nil means skip format-on-save in the current buffer.")

(defun mcg-format--supported-mode-p ()
  "Return non-nil when current `major-mode' supports format-on-save."
  (memq major-mode mcg-format-on-save-modes))

(defun mcg-format--lsp-available-p ()
  "Return non-nil when lsp-bridge can format the current buffer."
  (and mcg-format-prefer-lsp
       (bound-and-true-p lsp-bridge-mode)
       (fboundp 'lsp-bridge-has-lsp-server-p)
       (lsp-bridge-has-lsp-server-p)
       (fboundp 'lsp-bridge-code-format)))

(defun mcg-format--command-for-mode ()
  "Return formatter function for current `major-mode'."
  (cdr (assq major-mode mcg-format-command-alist)))

(defun mcg-format--external-candidates ()
  "Return external formatter candidates for current `major-mode'."
  (cdr (assq major-mode mcg-format-external-command-alist)))

(defun mcg-format--select-external-command ()
  "Return the first available external formatter command for current buffer."
  (cl-find-if (lambda (candidate)
                (and (consp candidate)
                     (executable-find (car candidate))))
              (mcg-format--external-candidates)))

(defun mcg-format--call-process (program args file)
  "Run PROGRAM with ARGS against FILE and return non-nil on success."
  (let ((buffer (get-buffer-create "*MCG Format*")))
    (with-current-buffer buffer
      (erase-buffer))
    (let ((exit-code
           (apply #'call-process program nil buffer nil
                  (append args (list file)))))
      (if (zerop exit-code)
          t
        (display-buffer buffer)
        (mcg-log-warning "Formatter failed: %s %s" program (string-join args " "))
        nil))))
#+end_src

** Commands

#+begin_src emacs-lisp
;;;###autoload
(defun mcg-format-with-external-command ()
  "Format current file with the first available external formatter."
  (interactive)
  (unless buffer-file-name
    (user-error "Current buffer is not visiting a file"))
  (let ((candidate (mcg-format--select-external-command)))
    (unless candidate
      (user-error "No formatter found for %s" major-mode))
    (save-buffer)
    (when (mcg-format--call-process (car candidate) (cdr candidate) buffer-file-name)
      (revert-buffer :ignore-auto :noconfirm)
      (mcg-log "Formatted %s with %s" buffer-file-name (car candidate))
      t)))

;;;###autoload
(defun mcg-format-buffer ()
  "Format current buffer using the best available formatter."
  (interactive)
  (cond
   ((mcg-format--lsp-available-p)
    (call-interactively #'lsp-bridge-code-format)
    t)
   ((let ((formatter (mcg-format--command-for-mode)))
      (when (and formatter (fboundp formatter))
        (call-interactively formatter)
        t)))
   (t
    (user-error "No formatter available for %s" major-mode))))

(defun mcg-format-buffer-maybe ()
  "Format current buffer before saving when enabled and supported."
  (when (and mcg-format-on-save
             (not mcg-format--inhibit-on-save)
             (mcg-format--supported-mode-p)
             (not buffer-read-only))
    (let ((mcg-format--inhibit-on-save t))
      (ignore-errors
        (mcg-format-buffer)))))

;;;###autoload
(defun mcg-format-toggle-on-save ()
  "Toggle global MCG format-on-save."
  (interactive)
  (setq mcg-format-on-save (not mcg-format-on-save))
  (message "MCG format-on-save %s" (if mcg-format-on-save "enabled" "disabled")))
#+end_src

** Initialize

#+begin_src emacs-lisp
(add-hook 'before-save-hook #'mcg-format-buffer-maybe)

(mcg-bind-key "C-c r f" #'mcg-format-buffer "editor/format")
(mcg-bind-key "C-c r F" #'mcg-format-toggle-on-save "editor/format")

(mcg-log "format module configured")
