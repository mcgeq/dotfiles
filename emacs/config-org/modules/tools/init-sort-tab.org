** Setup

#+begin_src emacs-lisp
(defun mcg--setup-sort-tab ()
  "Setup sort-tab."
  (when (mcg-require-extension "sort-tab" t)
    (when (fboundp 'sort-tab-mode)
      (sort-tab-mode 1))
    (mcg-log "sort-tab configured")))
#+end_src

** Keybindings

#+BEGIN_SRC emacs-lisp
  (defun mcg--setup-sort-tab-keybinding ()
    "Setup sort-tab keybinding."
    (when (featurep 'sort-tab)
      (mcg-bind-key "C-c t n" 'sort-tab-select-next-tab "tools/sort-tab")
      (mcg-bind-key "C-c t p" 'sort-tab-select-prev-tab "tools/sort-tab")
      (mcg-bind-key "M-8" 'sort-tab-select-next-tab "tools/sort-tab")
      (mcg-bind-key "M-7" 'sort-tab-select-prev-tab "tools/sort-tab")
      (mcg-bind-key "C-c t k" 'sort-tab-close-current-tab "tools/sort-tab")
      (mcg-bind-key "C-c t o" 'sort-tab-close-other-tabs "tools/sort-tab")
      (mcg-bind-key "C-c t f" 'sort-tab-select-first-tab "tools/sort-tab")
      (mcg-bind-key "C-c t l" 'sort-tab-select-last-tab "tools/sort-tab")
      )
    (mcg-log "Sort tab keybindings configured"))
#+END_SRC

** Initialize

#+begin_src emacs-lisp

  (mcg--setup-sort-tab-keybinding)
