** Helpers

#+begin_src emacs-lisp
;;;###autoload
(defun mcg-project-current-root (&optional maybe-prompt)
  "Return current project root.
When MAYBE-PROMPT is non-nil, ask the user to select a project if needed."
  (when-let* ((project (project-current maybe-prompt)))
    (file-name-as-directory (project-root project))))

(defun mcg-project--require-root ()
  "Return current project root or raise a user error."
  (or (mcg-project-current-root)
      (user-error "Not inside a project")))
#+end_src

** Commands

#+begin_src emacs-lisp
;;;###autoload
(defun mcg/project-switch ()
  "Switch project."
  (interactive)
  (call-interactively #'project-switch-project))

;;;###autoload
(defun mcg/project-find-file ()
  "Find a file in the current project."
  (interactive)
  (call-interactively #'project-find-file))

;;;###autoload
(defun mcg/project-switch-buffer ()
  "Switch to a buffer in the current project."
  (interactive)
  (if (fboundp 'project-switch-to-buffer)
      (call-interactively #'project-switch-to-buffer)
    (call-interactively #'switch-to-buffer)))

;;;###autoload
(defun mcg/project-dired ()
  "Open current project root in Dired."
  (interactive)
  (if (fboundp 'project-dired)
      (call-interactively #'project-dired)
    (dired (mcg-project--require-root))))

;;;###autoload
(defun mcg/project-eshell ()
  "Open Eshell in current project."
  (interactive)
  (if (fboundp 'project-eshell)
      (call-interactively #'project-eshell)
    (let ((default-directory (mcg-project--require-root)))
      (eshell t))))

;;;###autoload
(defun mcg/project-search ()
  "Search text in current project.
Prefer `consult-ripgrep' when available; otherwise use `project-find-regexp'."
  (interactive)
  (let ((root (mcg-project--require-root)))
    (cond
     ((fboundp 'consult-ripgrep)
      (consult-ripgrep root))
     ((fboundp 'project-find-regexp)
      (call-interactively #'project-find-regexp))
     (t
      (user-error "No project search command available")))))

;;;###autoload
(defun mcg/project-replace ()
  "Run query replace across current project."
  (interactive)
  (if (fboundp 'project-query-replace-regexp)
      (call-interactively #'project-query-replace-regexp)
    (user-error "project-query-replace-regexp is unavailable")))

;;;###autoload
(defun mcg/project-kill-buffers ()
  "Kill buffers belonging to current project."
  (interactive)
  (if (fboundp 'project-kill-buffers)
      (call-interactively #'project-kill-buffers)
    (user-error "project-kill-buffers is unavailable")))

;;;###autoload
(defun mcg/project-root ()
  "Show current project root."
  (interactive)
  (message "%s" (mcg-project--require-root)))

;;;###autoload
(defun mcg/project-build ()
  "Build current project.
Prefer `mcg-compile-build' when the compile module is loaded."
  (interactive)
  (cond
   ((fboundp 'mcg-compile-build)
    (call-interactively #'mcg-compile-build))
   ((fboundp 'project-compile)
    (call-interactively #'project-compile))
   (t
    (call-interactively #'compile))))

;;;###autoload
(defun mcg/project-test ()
  "Test current project.
Prefer `mcg-compile-test' when the compile module is loaded."
  (interactive)
  (if (fboundp 'mcg-compile-test)
      (call-interactively #'mcg-compile-test)
    (user-error "No project test command available")))
#+end_src
