** Initialize

#+begin_src emacs-lisp
;;; ============================================================
;;; Initialize
;;; ============================================================

(defun mcg--setup-majutsu ()
  "Setup majutsu - Magit for Jujutsu."
  (if (mcg-require-extension 'majutsu t)
      (progn
        ;; Setup keybindings
        (mcg-bind-key mcg-tools-majutsu-prefix 'majutsu "tools/majutsu")
        
        ;; Optional: Setup evil integration
        (when (and mcg-tools-majutsu-enable-evil-integration
                   (featurep 'evil))
          (require 'majutsu-evil nil t))
        
        (mcg-log "Majutsu configured with keybinding %s" mcg-tools-majutsu-prefix))
    (mcg-log-error "Failed to load Majutsu - ensure jj is installed")))

;; Setup majutsu after module load
(mcg--setup-majutsu)
#+end_src
