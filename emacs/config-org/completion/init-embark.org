** Consult Configuration
#+BEGIN_SRC emacs-lisp
  ;;; === Consult - 咨询式搜索命令 ===
  ;; Consult provides practical commands based on Emacs completion
  
  ;; Preview configuration
  (setq consult-preview-key 'any)          ; Enable preview for all commands
  (setq consult-narrow-key "<")            ; Narrow prefix key
  (setq consult-widen-key ">")             ; Widen key
  
  ;; Preview delay (nil for instant preview)
  (setq consult-preview-max-size 10485760) ; 10 MB max file size for preview
  (setq consult-preview-max-count 10)      ; Max number of preview buffers
  
  ;; Async commands configuration
  (setq consult-async-refresh-delay 0.15)  ; Delay before async search starts
  (setq consult-async-input-throttle 0.2)  ; Throttle input for async commands
  (setq consult-async-input-debounce 0.1)  ; Debounce async input
  
  ;; === 已移除 consult-ripgrep 和 consult-find 配置 ===
  ;; 这些功能已被 color-rg 和 blink-search 替代
  ;; - consult-ripgrep → color-rg-search-input (C-c s g)
  ;; - consult-find → blink-search (C-c s D/P)
  ;; 
  ;; 优势：
  ;; - 避免 Windows 中文编码问题
  ;; - 更好的性能和用户体验
  ;; - 立即显示结果，无需等待
  
  ;; === Windows UTF-8 全局配置（保留用于其他工具）===
  (when (eq system-type 'windows-nt)
    ;; 确保全局 UTF-8 编码
    (setq locale-coding-system 'utf-8)
    (set-default-coding-systems 'utf-8)
    (prefer-coding-system 'utf-8)
    
    ;; 进程编码配置（用于 grep、rg 等）
    (setq process-coding-system-alist
          (append '(("grep" . (utf-8 . utf-8))
                    ("rg" . (utf-8 . utf-8)))
                  process-coding-system-alist)))
#+END_SRC

  ;;; === 辅助函数 ===
  
  (defun mcg/embark-which-key-indicator ()
    "Show embark actions using which-key."
    (lambda (&optional keymap targets prefix)
      (if (not (and keymap targets))
          (which-key--hide-popup-ignore-command)
        (which-key--show-keymap
         (if (eq (plist-get (car targets) :type) 'embark-become)
             "Become"
           (format "Act on %s '%s'%s"
                   (plist-get (car targets) :type)
                   (embark--truncate-target
                    (plist-get (car targets) :target))
                   (if (cdr targets) ".." "")))
         (if prefix
             (pcase (lookup-key keymap prefix 'accept-default)
               ((and (pred keymapp) km) km)
               (_ (key-binding prefix 'accept-default)))
           keymap)
         nil nil t (lambda (binding)
                     (not (string-suffix-p "-argument" (cdr binding))))))))
