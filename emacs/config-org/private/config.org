* Private Configuration

This file is used to override default configuration. Loaded after all modules.

** LSP Bridge Configuration

#+begin_src emacs-lisp
;;; ============================================================
;;; LSP Bridge Configuration
;;; ============================================================

;; Enable inlay hints
(setq mcg-lsp-enable-inlay-hint t)

#+end_src

** Clutch Database Configuration

#+begin_src emacs-lisp
;;; ============================================================
;;; Clutch Database Configuration
;;; ============================================================

(setq mcg-tools-clutch-connections
      '(("dev-mysql" . (:backend mysql
                          :host "127.0.0.1" :port 3306
                          :user "root" :password "test"
                          :database "mydb"
                          :connect-timeout 5
                          :read-idle-timeout 60))
        ("dev-pg" . (:backend pg
                         :host "127.0.0.1" :port 5432
                         :user "morado" :password "morado"
                         :database "xiaolian"))
        ("dev-sqlite" . (:backend sqlite
                            :database "/path/to/my.db"))))

#+end_src

** Provide

#+begin_src emacs-lisp
(provide 'private-config)

#+end_src
